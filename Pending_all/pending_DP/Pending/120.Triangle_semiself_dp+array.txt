class Solution {
    public int minimumTotal(List<List<Integer>> t) {
        int len = t.size();
        int[] res = new int[len];
        for (int i = 0; i < len; i++) res[i] = t.get(len - 1).get(i);
        for (int i = len - 2; i >= 0; i--)
            for (int j = 0; j <= i; j++)
                res[j] = Math.min(res[j + 1], res[j]) + t.get(i).get(j);
        return res[0];
    }
}
